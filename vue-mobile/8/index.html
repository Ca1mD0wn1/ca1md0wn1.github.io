# 8.2 嵌套路由

一些应用程序的 UI 由多层嵌套的组件组成。在这种情况下，URL 的片段通常对应于特定的嵌套组件结构，例如：

/user/johnny/profile                     /user/johnny/posts
+------------------+                  +-----------------+
| User             |                  | User            |
| +--------------+ |                  | +-------------+ |
| | Profile      | |  +------------>  | | Posts       | |
| |              | |                  | |             | |
| +--------------+ |                  | +-------------+ |
+------------------+                  +-----------------+

通过 Vue Router，你可以使用嵌套路由配置来表达这种关系。

# 8.7 实现注册第一步功能

<!-- src/views/regigster/components/CheckTelComponent.vue -->
<template>
  <div class="box">
    <van-field v-model="tel" placeholder="请输入手机号" />
    <van-button block round :disabled="flag" color="linear-gradient(to right, #ff6034, #ee0a24)" @click="next">
      下一步
    </van-button>
  </div>
</template>
<script lang="ts">
  import { defineComponent } from 'vue'
  import { Dialog } from 'vant'
  import { doCheckPhone } from '@/api/user'
  export default defineComponent({
    data () {
      return {
        tel: ''
      }
    },
    computed: {
      flag () {
        return !/^(?:(?:\+|00)86)?1[3-9]\d{9}$/.test(this.tel)
      }
    },
    methods: {
      next () {
        doCheckPhone({ tel: this.tel }).then(res => {
          if (res.data.code === '10005') {
            // 手机号已被注册，提醒用户，是否直接登录
            Dialog.confirm({
              message: '该手机号已被注册，请直接登录',
              confirmButtonText: '立即登录',
              cancelButtonText: '取消'
            })
              .then(() => {
                // on confirm
                // 登录 - 注册 - 直接返回上一页即可
                this.$router.back()
              })
              .catch(() => {
                // on cancel
              })
          } else {
            // 未被注册 跳转到 发送短信验证码页面
            localStorage.setItem('tel', this.tel)
            this.$router.push('/register/sms')
          }
        })
      }
    }
  })
</script>

# 8.8 实现注册第二步功能

<!-- src/views/regigster/components/SendTelCodeComponent.vue -->
<template>
  <div class="box">
    <div class="tip">
      <van-image :src="sms" height="71"></van-image>
      <p>我们将以短信形式将验证码发送给您</p>
    </div>
    <van-field
      v-model="telcode"
      clearable
      placeholder="请输入短信验证码"
    >
      <template #button>
        <van-button class="codeBtn" round size="small" type="primary" @click="sendCode">发送验证码</van-button>
      </template>
    </van-field>
    <van-button class="nextBtn" block round :disabled="flag" color="linear-gradient(to right, #ff6034, #ee0a24)" @click="next">
      下一步
    </van-button>
  </div>
</template>
<script lang="ts">
  import { defineComponent } from 'vue';
  import { Toast } from 'vant'
  import sms from '@/assets/sms.png'
  import { doSendMsgCode, doCheckCode } from '@/api/user'
  export default defineComponent({
    data () {
      return {
        tel: localStorage.getItem('tel'),
        telcode: '',
        sms
      }
    },
    computed: {
      flag () {
        return !/^\d{5,}$/.test(this.telcode)
      }
    },
    mounted () {
      // 如果用户直接在地址栏输入第二步，有手机号继续，没有手机号替换到第一页
      if (!localStorage.getItem('tel')) {
        this.$router.replace('/register/index')
      }
    },
    methods: {
      sendCode () {
        // 倒计时效果在此处？ √
        doSendMsgCode({ tel: this.tel! }).then(res => {
          // 正常此时手机中已经可以收到短信验证码（也可能收不到，但是跟前端无关）
          // 倒计时效果在此处？ ×
          // 本接口文档验证码没有给手机发送，验证码在返回值中，可以通过控制台查看
          console.log(res.data) // 真实情况 这里不做任何的打印
        })
      },
      next () {
        doCheckCode({ tel: this.tel!, telcode: this.telcode }).then(res => {
          if (res.data.code === '10007') {
            // 验证码错误 提醒用户
            Toast('验证码错误')
          } else {
            this.$router.push('/register/pwd')
          }
        })
      }
    }
  })
</script>

# 8.9 实现注册第三步功能

<!-- src/views/regigster/components/SetPasswordComponent.vue -->
<template>
  <div class="box">
    <div class="tip">
      <van-image :src="pwd" height="71"></van-image>
      <p>请设置登录密码</p>
    </div>
    <van-field :type="type" v-model="password" clearable placeholder="请设置至少6位登录密码" :right-icon="eye" @click-right-icon="changeType" />
    <p class="pwdTip">密码强度校验，最少6位，包括至少1个大写字母，1个小写字母，1个数字</p>
    <van-button block round :disabled="flag" color="linear-gradient(to right, #ff6034, #ee0a24)" @click="next">
      完成
    </van-button>
  </div>
</template>
<script lang="ts">
  import { defineComponent } from 'vue';
  import pwd from '@/assets/pwd.png'
  import type { FieldType } from 'vant';

  import { doFinishRegister } from '@/api/user'
  interface IData {
    tel: string
    password: string
    pwd: string
    type: FieldType
  }
  export default defineComponent({
    data (): IData {
      return {
        tel: localStorage.getItem('tel')!,
        password: '',
        pwd,
        type: 'password'
      }
    },
    computed: {
      flag (): boolean {
        return !/^\S*(?=\S{6,})(?=\S*\d)(?=\S*[A-Z])(?=\S*[a-z])\S*$/.test(this.password)
      },
      eye (): string {
        return this.type === 'password' ? 'closed-eye' : 'eye-o'
      }
    },
    methods: {
      next (): void {
        doFinishRegister({ tel: this.tel, password: this.password }).then(res => {
          // 注册成功，成功之后呢？ --- 回到登录
          // 登录 - 注册第一步 - 注册第二步 - 注册第三步
          localStorage.removeItem('tel')
          this.$router.go(-3)
        })
      },
      changeType () {
        if (this.type === 'password') {
          this.type = 'text'
        } else {
          this.type = 'password'
        }
      }
    }
  })
</script>
