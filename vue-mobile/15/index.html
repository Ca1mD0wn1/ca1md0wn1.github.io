5.18 状态管理器

vuex： https://vuex.vuejs.org/zh/

5.18.1 介绍

Vuex 是一个专为 Vue.js 应用程序开发的状态管理模式 + 库。它采用集中式存储管理应用的所有组件的状态，并以相应的规则保证状态以一种可预测的方式发生变化。

5.18.2 什么时候使用状态管理器

  • 多个视图依赖于同一状态。1
  • 来自不同视图的行为需要变更同一状态。

登录状态：首页头部，加入购物车，购物车，确认订单，地址管理。。。。（登录，退出）

购物车数量：购物车，页面底部，详情页

城市选择：首页头部，城市选择列表

// src/store/index.ts
// 1.引入创建仓库的函数
import { createStore, Store } from 'vuex'

// 2.创建仓库
const store: Store<{}> = createStore({
  state: { // vuex管理的状态
    count: 1
  },
  getters: { // vuex的计算属性
    doubleCount (state) {
      return state.count * 2
    }
  },
  actions: { // 异步操作
    addAction (context) { // context 上下文对象
      setTimeout(() => {
        context.commit('increment', 10) // 交给mutations改变状态
      }, 3000)
    },
    addParamsAction (context, payload) { // context 上下文对象
      setTimeout(() => {
        context.commit('increment', payload) // 交给mutations改变状态
      }, 3000)
    }
  },
  mutations: { // 同步更改状态
    increment (state, payload) {
      state.count += payload
    }
  },
  modules: { // 分模块管理状态

  }
})

// 3.暴露仓库
export default store

<!-- src/views/vuex/IndexView.vue -->
<template>
  <div>
    {{ count }} - {{ doubleCount }}
    <button @click="add">加1</button>
    <button @click="addVuexFn">vuex中异步加10</button>
    <button @click="addVuexParamsFn">vuex中异步加5-传递参数</button>
    <button @click="addComFn">组件中异步加100</button>
  </div>
</template>

<script lang="ts">
  import { defineComponent } from 'vue';

  export default defineComponent({
    mounted () {
    },
    computed: {
      count () {
        return this.$store.state.count
      },
      doubleCount () {
        return this.$store.getters.doubleCount
      }
    },
    methods: {
      add () { // 组件同步更改state
        this.$store.commit('increment', 1)
      },
      addVuexFn () { // 组件异步交给vuex的actions解决异步操作，操作结束，同步修改state
        this.$store.dispatch('addAction')
      },
      addVuexParamsFn () {// 组件异步交给vuex的actions解决异步操作,并且传递参数，操作结束，同步修改state
        this.$store.dispatch('addParamsAction', 5)
      },
      addComFn () { // 组件异步处理完成同步修改state
        setTimeout(() => {
          this.$store.commit('increment', 100)
        }, 3000)
      }
    }
  })
</script>

<!-- src/views/vuex/IndexView.vue -->
<template>
  <div>
    {{ count }} - {{ doubleCount }}
    <button @click="add">加1</button>
    <button @click="addVuexFn">vuex中异步加10</button>
    <button @click="addVuexParamsFn">vuex中异步加5-传递参数</button>
    <button @click="addComFn">组件中异步加100</button>
  </div>
</template>

<script lang="ts">
  import { defineComponent } from 'vue';
  import { mapState, mapGetters, mapMutations, mapActions } from 'vuex'
  export default defineComponent({
    mounted () {
    },
    computed: { // mapState mapGetters 和计算属性一起使用
      // count () {
      //   return this.$store.state.count
      // },
      // doubleCount () {
      //   return this.$store.getters.doubleCount
      // }
      ...mapState({ // 获取状态
        count: (state: any) => state.count
      }),
      ...mapGetters({ // 获取计算属性
        doubleCount: 'doubleCount' // 后面的 doubleCount 相当于 this.$store.getters.doubleCount
      })
    },
    methods: { // mapMutations 和  mapActions 与 自定义函数一起使用
      ...mapMutations({ // 前面的increment 相当于在当前组件定义了一个函数，可以使用参数，参数即为使用 this.$store.commit('increment', 1) 使用的参数
        increment: 'increment' // 后面的increment 相当于 this.$store.commit('increment')
      }),
      ...mapActions({
        addAction: 'addAction', // this.$store.dispatch('addAction')
        addParamsAction: 'addParamsAction' //  this.$store.dispatch('addParamsAction', 5)
      }),
      add () { // 组件同步更改state
        // this.$store.commit('increment', 1)
        this.increment(1)
      },
      addVuexFn () { // 组件异步交给vuex的actions解决异步操作，操作结束，同步修改state
        // this.$store.dispatch('addAction')
        this.addAction()
      },
      addVuexParamsFn () {// 组件异步交给vuex的actions解决异步操作,并且传递参数，操作结束，同步修改state
        // this.$store.dispatch('addParamsAction', 5)
        this.addParamsAction(5)
      },
      addComFn () { // 组件异步处理完成同步修改state
        setTimeout(() => {
          // this.$store.commit('increment', 100)
          this.increment(100)
        }, 3000)
      }
    }
  })
</script>

5.18.5 模块化开发

假设有一个产品模块，产品模块要管理自己的状态 actions 以及mutations

import type { IBannerItem, IPro } from "@/views/home/home"
import { getBannerListData } from '@/api/home'
import type {Context } from "vm"
// src/store/pro.ts
interface IState {
  bannerList: IBannerItem[]
  proList: IPro[]
}
export default {
  namespaced: true, // 命名空间
  state: { 
    bannerList: [], // state.pro.bannerList
    proList: [] // state.pro.proList
  },
  actions: { // 这样设计意味着 轮播图异步在vuex，而 产品异步在组件
    getBannerListAction (context: Context) { // this.$store.dispatch('pro/getBannerListAction')
      getBannerListData().then(res => {
        context.commit('changeBannerList', res.data.data)
      })
    }
  },
  mutations: {
    changeBannerList (state: IState, payload: IBannerItem[]) {
      state.bannerList = payload
    },
    changeProList (state: IState, payload: IPro[]) { // this.$store.commit('pro/changeProList')
      state.proList = payload
    }

  }
}
