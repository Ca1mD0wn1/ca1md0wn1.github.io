数组扁平化

const arr = [1, [2, [3, [4, 5]]], 6]


function reduceDimension(arr) {
  let ret = [];
  let toArr = function (arr) {
    arr.forEach(function (item) {
      item instanceof Array ? toArr(item) : ret.push(item);
    });
  }
  toArr(arr);
  return ret;

}
console.log(reduceDimension(arr))

输出结果:[ 1, 2, 3, 4, 5, 6 ]

function fn(str) {
  let zhan = []
  for (let i = 0; i < str.length; i++) {
    if (str[i] === "(" || str[i] === "[" || str[i] === "{") { zhan.push(str[i]) } else {
      if (str[i] === "}" && "{" === zhan.pop()) { continue } if (str[i] === ")" && "(" === zhan.pop()) { continue } if (str[i] === "]"
        && "[" === zhan.pop()) { ; continue } return false;
    }
  } return zhan.length > 0 ? false : true;
}

分糖果问题:


function candy2(arr) {
  let res = new Array(arr.length).fill(1)
  for (let i = 0; i < arr.length; i++) {
    if (arr[i + 1] > arr[i]) {
      res[i + 1] = res[i] + 1
    }
  }
  for (let j = arr.length - 1; j > 0; j--) {
    res[j - 1] = Math.max(res[j - 1], res[j] + 1)
    res[j - 1] = res[j] + 1
  }

  return res.reduce((a, b) => a + b);
}

console.log(candy2([1, 1, 2]));//4

兔子问题:（斐波那契数列）

function rabbit(n) {
  if (n <= 0)
    return 0;
  if (n == 1 || n == 2)
    return 1;
  return rabbit(n - 1) + rabbit(n - 2);
}

console.log(
  rabbit(6)
)
